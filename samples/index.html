require_once 'vendor/autoload.php';

$classes = [
'تقویتی',
'کنکور'
];

Fast(
function ($update, $obj) use ($classes) {
foreach ($update as $message) {

$message = isset($message['message']) ? $message['message'] : $message;
$text = $message['text'];
$user_id = $message['author_object_guid'];

$req = explode(' ', $text); // message format : شرکت در کلاس [$CLASSES] [$PHONE or $EMAIL]


if ($req[0] == 'ثبت نام در کلاس' && in_array($req[1], $classes)) {

$requests = json_decode(file_get_contents('request.json'), true);
$requests[] = "{$req[2]}";

file_put_contents('request.json', json_encode($requests));

$obj->sendMessage($user_id, 'درخواست شما ثبت شد.');
}

}
},
9123456789 // your phone
);
